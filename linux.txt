# Linux Commands Reference for DevOps

## System Information
ls -la                     # List all files with details including hidden files
pwd                       # Print working directory
whoami                    # Show current user
uname -a                  # Display system information
df -h                     # Show disk space usage in human-readable format
free -h                   # Display memory usage
top                       # Show running processes (interactive)
htop                      # Enhanced version of top (if installed)
ps aux                    # List all running processes
systemctl status         # Check status of system services

## File Operations
cp -r source dest        # Copy files/directories recursively
mv source dest           # Move/rename files or directories
rm -rf dir              # Force remove directory and contents
chmod +x file           # Make file executable
chown user:group file   # Change file ownership
tar -czvf archive.tar.gz dir/  # Create compressed archive
tar -xzvf archive.tar.gz      # Extract compressed archive
find / -name filename    # Search for files in entire system
grep -r "pattern" .     # Search for text recursively
sed -i 's/old/new/g' file  # Replace text in files

## Network
netstat -tulpn          # Show active network connections
ss -tulpn              # Modern alternative to netstat
curl -X GET url        # Make HTTP request
wget url               # Download files
ping host             # Test network connectivity
nslookup domain       # DNS lookup
ip addr show          # Show network interfaces
iptables -L           # List firewall rules
nc -zv host port      # Test port connectivity
tcpdump -i eth0       # Capture network traffic

## Process Management
ps aux | grep process  # Find specific process
kill -9 PID           # Force kill process
killall processname   # Kill all processes by name
nohup command &       # Run command in background
jobs                  # List background jobs
fg                    # Bring job to foreground
bg                    # Send job to background
nice -n 19 command    # Run command with low priority

## Package Management (Ubuntu/Debian)
apt update            # Update package list
apt upgrade           # Upgrade installed packages
apt install package   # Install package
apt remove package    # Remove package
dpkg -i package.deb   # Install .deb package

## System Services
systemctl start service    # Start service
systemctl stop service     # Stop service
systemctl restart service  # Restart service
systemctl enable service   # Enable service at boot
systemctl disable service  # Disable service at boot
journalctl -u service     # View service logs

## User Management
useradd username         # Create new user
usermod -aG group user   # Add user to group
passwd username         # Set/change password
su - username          # Switch user
sudo command           # Run command as root

## File Permissions
chmod 755 file         # Set rwx for owner, rx for others
chmod u+x file        # Add execute permission for owner
chown user:group file # Change file ownership
umask 022            # Set default permissions

## Monitoring and Logs
tail -f /var/log/syslog  # Watch system logs live
dmesg                    # Show kernel messages
iostat                   # Show I/O statistics
vmstat                   # Show virtual memory stats
sar                      # System activity reporter
lsof                     # List open files

## Docker-specific
docker ps              # List running containers
docker images          # List images
docker build -t name . # Build image from Dockerfile
docker-compose up -d   # Start services in background
docker logs container  # View container logs

## Kubernetes-specific
kubectl get pods       # List pods
kubectl get nodes      # List nodes
kubectl describe pod   # Show pod details
kubectl logs pod       # View pod logs
kubectl apply -f file  # Apply configuration

## Storage
lsblk                 # List block devices
fdisk -l              # List disk partitions
mount                 # Show mounted filesystems
dd if=/dev/zero of=file bs=1M count=100  # Create test file

## Performance
uptime               # Show system load
stress              # Stress test the system
ulimit -a           # Show resource limits
nice -n 19 command  # Run with low priority
ionice -c2 -n7 command  # Set I/O priority
